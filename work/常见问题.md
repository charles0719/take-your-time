# 如何实现 MySQL 删除重复记录并且只保留一条？

```sql
UPDATE
 dev_biztpl_group_article
set DELFLAG = 1
 -- where article_id = '8a209d1f7627b7d701764a7d048e0582'
WHERE
 id NOT IN (
  SELECT
   dt.minno
  FROM
   (
    SELECT
     MIN(id) AS minno
    FROM
     dev_biztpl_group_article
		where article_id = '8a209d1f7627b7d701764a7d048e0582'
    GROUP BY
     GROUP_ID,ATTR_ID
   ) dt
 )
and article_id = '8a209d1f7627b7d701764a7d048e0582'

```
